\begin{frame}{}
  \begin{theorem}[``Weakness'' of Atomic Registers]
    Atomic read/write registers \red{\it cannot} solve $2$-thread consensus.
  \end{theorem}

  \pause
  \vspace{0.60cm}
  \begin{proof}
    \begin{enumerate}[(i)]
      \item By contradiction: $\exists \mathcal{P}$
      \pause
      \item Run $\mathcal{P}$ until it reaches a \blue{critical} state $s$.
    \end{enumerate}

    \pause
    \vspace{0.50cm}
    \centerline{\red{$Q:$ What are the next decision steps of two threads?}}
  \end{proof}
\end{frame}
