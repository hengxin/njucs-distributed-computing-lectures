  \pause
  \begin{definition}[The Consensus Problem]
    \begin{description}
      \setlength{\itemsep}{5pt}
      \item[Consistent] All \blue{\footnotesize (non-faulty)} threads must \red{\it agree} on the same value.
      \item[Valid] The common decision value must be the value \red{\it proposed} by some thread.
      \item[Wait-free] Each \blue{\footnotesize (non-faulty)} thread must decide in a \red{\it finite} number of steps.
    \end{description}
  \end{definition}
\end{frame}
